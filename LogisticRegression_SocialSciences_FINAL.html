#Predicting employment outcome based on field of study

Model 1

pred_1 <- predict(mod_1, test_1, type = "response") #predicting probabilities using test set

actual = as.numeric(as.character(test_1$LFSTATP)) #converting actual classes to numeric

# Create ROC curve
roc_curve <- roc(actual, pred_1)
## Setting levels: control = 0, case = 1
## Setting direction: controls < cases
# Plot ROC curve
plot(roc_curve, main = "ROC Curve")

#find best threshold
coordinate_thresh <- pROC::coords(roc_curve,"best", ret = "threshold")
optimal_threshold <- coordinate_thresh$threshold


pred_data = rep(0,nrow(test_1)) #Assign classes 
pred_data[pred_1 > optimal_threshold] = 1 #Assign classes based on optimal threshold
summary[1,2] = mean(actual==pred_data) #Prediction accuracy
conf.mat <- table(actual, pred_data) #Confusion Matrix 

#calculating precision
TP <- conf.mat[2,2]
FP <- conf.mat[1,2]
summary[1,3] <-  TP / (TP + FP)

#calculating recall
FN <- conf.mat[2,1]
summary[1,4] <-  TP / (TP + FN)

# AUC 
summary[1,5] <- pROC::auc(roc_curve)

#f1 
summary[1,6] <- 2*((summary[1,3]*summary[1,4])/(summary[1,3]+summary[1,4]))
